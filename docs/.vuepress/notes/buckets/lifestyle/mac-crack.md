## Beyond Compare 4.x [SP]

Install Command Line Tools:

    /usr/local/bin/bcomp:
    Launches comparison and waits for it to complete.

    /usr/local/bin/bcompare:
    Launches comparison and returns immediately.

.gitconfig Settings:：

    [core]
        excludesfile = /Users/gongyuqi/.gitignore_global
    [difftool "sourcetree"]
        cmd = /usr/local/bin/bcomp \"$LOCAL\" \"$REMOTE\"
        path = -ro
    [mergetool "sourcetree"]
        cmd = /usr/local/bin/bcomp \"$LOCAL\" \"$REMOTE\" \"$BASE\" \"$MERGED\"
        trustExitCode = true
    [user]
        name = yuqigong
        email =
    [commit]
        template = /Users/gongyuqi/.stCommitMsg
    [diff]
        tool = bc3
    [difftool]
        prompt = false
    [difftool]
        tool = bc3
    [merge]
        prompt = false
    [merge]
        tool = bc3
    [mergetool]
        bc3 = trustExitCode
